SELECT
                   st.style_code,
                   bd.name as brand_name,
                   bz.name as size_name,
                   st.per_packet_shocks_quantity,
                   st.remaining_socks,
                   st.individual_buy_price,
                   st.individual_sale_price 
               FROM
                   stocks st 
                   left join
                   brands bd 
                   on st.brand_id = bd.id 
                   left join
                   brand_sizes bz 
                   on st.brand_size_id = bz.id 
               where 
                   remaining_socks > 0 and st.type_id = 6 and (status=0 or status=5) and st.sale_type = 2  and st.style_code not in ('')  order by st.lot_no,st.brand_id
